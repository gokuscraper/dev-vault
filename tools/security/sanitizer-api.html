🧩 Custom Element Handling

By default the Sanitizer API removes unknown (custom) elements. Use allowElements to permit them.

Shadow DOM Behavior

The Sanitizer API processes <template shadowrootmode="open"> declarative shadow roots. Content inside shadow roots is sanitized recursively with the same config. Scripts and event handlers inside shadow DOM are removed just like in the light DOM. 

// Shadow DOM content is also sanitized:
const input = `<div>
  <template shadowrootmode="open">
    <style>:host { color: red; }</style>
    <p onclick="alert('xss')">Shadow content</p>
  </template>
</div>`;

// After sanitization:
// <template> and shadow content are processed
// onclick handler is removed
// <style> may be removed (depending on config)

// Option 1: DOMPurify as Sanitizer polyfill
// DOMPurify provides similar functionality
// and works in all browsers
import DOMPurify from 'dompurify';

function sanitizeHTML(dirty) {
  if ('Sanitizer' in window) {
    const s = new Sanitizer();
    const el = document.createElement('div');
    el.setHTML(dirty, { sanitizer: s });
    return el.innerHTML;
  }
  return DOMPurify.sanitize(dirty);
}

      

      

        
        
// Option 2: Feature detection + graceful fallback
function safeInsertHTML(element, html, config) {
  if (element.setHTML) {
    // Native Sanitizer API
    element.setHTML(html, config
      ? { sanitizer: new Sanitizer(config) }
      : undefined
    );
  } else {
    // Fallback: parse + manual strip
    const doc = new DOMParser()
      .parseFromString(html, 'text/html');
    // Remove scripts, event handlers, etc.
    stripDangerous(doc.body);
    element.innerHTML = doc.body.innerHTML;
  }
}
