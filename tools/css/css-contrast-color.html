Strategy 1: @supports

/* Fallback: precomputed value */ .button { background: #6c8cff; color: #fff; /* fallback */ } /* Native when supported */ @supports (color: contrast-color(#000)) { .button { color: contrast-color(#6c8cff); } }

Strategy 2: JavaScript Feature Detection + Polyfill

function supportsContrastColor() { if (typeof CSS === 'undefined' || !CSS.supports) return false; return CSS.supports('color', 'contrast-color(#000)'); } function contrastColor(bgHex) { // Parse hex to RGB const r = parseInt(bgHex.slice(1, 3), 16); const g = parseInt(bgHex.slice(3, 5), 16); const b = parseInt(bgHex.slice(5, 7), 16); // WCAG relative luminance const [rs, gs, bs] = [r, g, b].map(c => { c = c / 255; return c <= 0.04045 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4); }); const L = 0.2126 * rs + 0.7152 * gs + 0.0722 * bs; return L > 0.179 ? '#000000' : '#ffffff'; } // Usage const bg = '#6c8cff'; const textColor = supportsContrastColor() ? `contrast-color(${bg})` : contrastColor(bg);

Strategy 3: CSS Custom Properties + JS Sync

:root { --bg-primary: #6c8cff; --text-primary: #ffffff; /* JS-computed fallback */ } @supports (color: contrast-color(#000)) { :root { --text-primary: contrast-color(var(--bg-primary)); } } .button { background: var(--bg-primary); color: var(--text-primary); }
