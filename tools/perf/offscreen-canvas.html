OffscreenCanvas + Worker Setup

Show
// Main thread: transfer canvas control to worker
const canvas = document.getElementById('myCanvas');
const offscreen = canvas.transferControlToOffscreen();

const worker = new Worker('worker.js');
worker.postMessage({ type: 'init', canvas: offscreen }, [offscreen]);

// Worker (worker.js):
self.onmessage = (e) => {
  if (e.data.type === 'init') {
    const canvas = e.data.canvas;
    const ctx = canvas.getContext('2d');

    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      // ... your rendering code ...
      requestAnimationFrame(draw);
    }
    draw();
  }
};

Blob URL Worker (Single-File Embedding)

Show
// Embed worker code as a Blob URL — no separate file needed
const workerCode = `
  self.onmessage = (e) => {
    const canvas = e.data.canvas;
    const ctx = canvas.getContext('2d');
    function animate() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      // render ...
      requestAnimationFrame(animate);
    }
    animate();
  };
`;

const blob = new Blob([workerCode], { type: 'application/javascript' });
const workerUrl = URL.createObjectURL(blob);
const worker = new Worker(workerUrl);

const offscreen = canvas.transferControlToOffscreen();
worker.postMessage({ canvas: offscreen }, [offscreen]);

Feature Detection & Fallback

Show
// Check if OffscreenCanvas is available
if (typeof OffscreenCanvas !== 'undefined') {
  // Use OffscreenCanvas in a worker
  const offscreen = canvas.transferControlToOffscreen();
  worker.postMessage({ canvas: offscreen }, [offscreen]);
} else {
  // Fallback: render on main thread
  const ctx = canvas.getContext('2d');
  function fallbackRender() {
    // ... main thread rendering ...
    requestAnimationFrame(fallbackRender);
  }
  fallbackRender();
}

// Also check transferControlToOffscreen support
const supportsTransfer = typeof HTMLCanvasElement.prototype
  .transferControlToOffscreen === 'function';
