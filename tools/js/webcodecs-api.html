API Code Example

// Create a VideoFrame from a canvas const frame = new VideoFrame(canvas, { timestamp: performance.now() * 1000, duration: 33333 // ~30fps in microseconds }); // Read raw pixel data from the frame const buf = new Uint8Array(frame.allocationSize()); await frame.copyTo(buf); // Apply pixel manipulation... for (let i = 0; i < buf.length; i += 4) { buf[i] = 255 - buf[i]; // R buf[i+1] = 255 - buf[i+1]; // G buf[i+2] = 255 - buf[i+2]; // B } // Create a new frame from modified data const processed = new VideoFrame(buf, { format: 'RGBA', codedWidth: frame.codedWidth, codedHeight: frame.codedHeight, timestamp: frame.timestamp }); frame.close();
