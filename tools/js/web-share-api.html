🛡️ Fallback Strategy

      

      

        
Always provide fallbacks when Web Share API is unavailable. Here is the recommended cascade.

        
    
          
  • 1️⃣Try navigator.share() — native dialog
    • 
          
  • 2️⃣Fall back to navigator.clipboard.writeText() — copy to clipboard
    • 
          
  • 3️⃣Show platform-specific share URLs (Twitter, Email, etc.)
    • 
          
  • 4️⃣Manual copy with document.execCommand('copy')
    • 
        


        
        
async function share(data) {
  if (navigator.share) {
    try {
      await navigator.share(data);
      return 'shared';
    } catch (e) {
      if (e.name === 'AbortError') return 'cancelled';
    }
  }
  // Fallback: copy to clipboard
  const text = `${data.title}\n${data.text}\n${data.url}`;
  if (navigator.clipboard?.writeText) {
    await navigator.clipboard.writeText(text);
    return 'copied';
  }
  // Last resort: prompt
  prompt('Copy this:', text);
  return 'manual';
}


        

          
        

        

      

    


    
    

      

        
🎯 Share Target Registration (PWA)

      

      

        
Register your PWA as a share target by adding share_target to your Web App Manifest.


        

          

            

              
              
            

            

              
              
            

            

              
              
            

            

              
              
            

            

              
              
            

            

              
              
            

            

              
              
            

            
          

          

            
            
{
  "share_target": {
    "action": "/share-receiver",
    "method": "POST",
    "enctype": "multipart/form-data",
    "params": {
      "title": "title",
      "text": "text",
      "url": "url",
      "files": [{
        "name": "media",
        "accept": ["image/*", "text/*"]
      }]
    }
  }
}
