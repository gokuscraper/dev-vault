Standard Video PiP

// Enter Picture-in-Picture const video = document.querySelector('video'); async function togglePiP() { if (document.pictureInPictureElement) { await document.exitPictureInPicture(); } else if (document.pictureInPictureEnabled) { const pipWindow = await video.requestPictureInPicture(); pipWindow.addEventListener('resize', (e) => { console.log(`PiP resized: ${pipWindow.width}x${pipWindow.height}`); }); } } video.addEventListener('enterpictureinpicture', (e) => { console.log('Entered PiP', e.pictureInPictureWindow); }); video.addEventListener('leavepictureinpicture', () => { console.log('Left PiP'); });

Canvas as PiP Source

const canvas = document.getElementById('myCanvas'); const stream = canvas.captureStream(30); // 30 FPS const video = document.createElement('video'); video.srcObject = stream; video.muted = true; await video.play(); const pipWindow = await video.requestPictureInPicture();

Document PiP (Chrome 116+)

// Document Picture-in-Picture const pipWindow = await documentPictureInPicture.requestWindow({ width: 400, height: 300 }); // Copy styles [...document.styleSheets].forEach(sheet => { try { const css = [...sheet.cssRules].map(r => r.cssText).join(''); const style = pipWindow.document.createElement('style'); style.textContent = css; pipWindow.document.head.appendChild(style); } catch(e) {} // cross-origin sheets }); // Add content const content = document.getElementById('player').cloneNode(true); pipWindow.document.body.appendChild(content); pipWindow.addEventListener('pagehide', () => { console.log('Document PiP closed'); });
