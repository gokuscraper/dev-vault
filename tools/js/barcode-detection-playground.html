Basic Detection

// Check browser support if ('BarcodeDetector' in window) { console.log('BarcodeDetector is supported!'); } // Check supported formats const formats = await BarcodeDetector.getSupportedFormats(); console.log('Supported formats:', formats); // Create detector (all formats) const detector = new BarcodeDetector(); // Or specific formats only const qrDetector = new BarcodeDetector({ formats: ['qr_code'] });

Detect from Image

const detector = new BarcodeDetector(); const img = document.querySelector('img'); const barcodes = await detector.detect(img); for (const barcode of barcodes) { console.log('Format:', barcode.format); console.log('Value:', barcode.rawValue); console.log('Bounding box:', barcode.boundingBox); console.log('Corner points:', barcode.cornerPoints); }

Live Camera Scanning

const video = document.querySelector('video'); const detector = new BarcodeDetector({ formats: ['qr_code', 'ean_13'] }); // Start camera const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } }); video.srcObject = stream; // Continuous scanning loop async function scanFrame() { try { const barcodes = await detector.detect(video); for (const barcode of barcodes) { console.log(`[${barcode.format}] ${barcode.rawValue}`); } } catch (e) { console.error('Detection failed:', e); } requestAnimationFrame(scanFrame); } video.addEventListener('loadeddata', scanFrame);

Draw Bounding Boxes

const canvas = document.querySelector('canvas'); const ctx = canvas.getContext('2d'); function drawDetections(barcodes) { ctx.clearRect(0, 0, canvas.width, canvas.height); for (const barcode of barcodes) { const { x, y, width, height } = barcode.boundingBox; // Draw bounding box ctx.strokeStyle = '#00ff88'; ctx.lineWidth = 2; ctx.strokeRect(x, y, width, height); // Draw corner points ctx.fillStyle = '#00ff88'; for (const point of barcode.cornerPoints) { ctx.beginPath(); ctx.arc(point.x, point.y, 4, 0, Math.PI * 2); ctx.fill(); } // Draw label ctx.fillStyle = '#00ff88'; ctx.font = '14px monospace'; ctx.fillText( `${barcode.format}: ${barcode.rawValue}`, x, y - 6 ); } }
