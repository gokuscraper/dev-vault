const anim = el.animate([
  { transform: 'translateX(0px)' },
  { transform: 'translateX(500px)', background: '#3fb950' }

anim.play();          // Start
anim.pause();         // Pause

Composite Modes

When multiple animations target the same property, composite controls how values combine.

replace default

Second animation overwrites the first entirely.

replace

add

Values are added on top of the underlying value.

add

accumulate

Values are accumulated (combined arithmetically).

accum

Code

// Base animation on all three
el.animate([
  { transform: 'translateX(0)' },
  { transform: 'translateX(100px)' }
], { duration: 1500, fill: 'forwards' });

// Second animation with different composite modes
el.animate([
  { transform: 'translateY(0)' },
  { transform: 'translateY(-40px)' }
], { duration: 1500, fill: 'forwards', composite: 'replace' });
// replace: only translateY applies (translateX gone)

el.animate([...], { composite: 'add' });
// add: translateX(100px) translateY(-40px) — both apply

el.animate([...], { composite: 'accumulate' });
// accumulate: values combine arithmetically

Animation Groups & Sequences

WAAPI doesn't have native GroupEffect/SequenceEffect yet (spec draft), but we can orchestrate them with delays and promises.

Parallel (Group)

B
C

Sequence (Chained)

1
2
3

Code

// Parallel: all start at the same time
const anims = elements.map(el =>
  el.animate(keyframes, { duration: 800, fill: 'forwards' })
);
// Wait for all to finish
await Promise.all(anims.map(a => a.finished));

// Sequence: chain with delay offsets
async function sequence(elements, keyframes, opts) {
  for (const el of elements) {
    const anim = el.animate(keyframes, opts);
    await anim.finished; // wait before next
  }
}

// Or use delay offsets for overlap
elements.forEach((el, i) => {
  el.animate(keyframes, {
    ...opts,
    delay: i * 300  // stagger by 300ms
  });
});

WAAPI vs CSS Animations

Same visual animation, two different approaches. WAAPI gives you runtime control CSS cannot.

WAAPI (JavaScript)
JS
CSS @keyframes
CSS

WAAPI Code

const anim = el.animate([
  { transform: 'translateY(0) scale(1)',
    background: '#58a6ff' },
  { transform: 'translateY(-50px) scale(1.2)',
    background: '#bc8cff' },
  { transform: 'translateY(0) scale(1)',
    background: '#58a6ff' }
], {
  duration: 1500,
  easing: 'ease-in-out',
  iterations: Infinity
});

// Runtime control:
anim.pause();
anim.play();
anim.reverse();
anim.playbackRate = 2;
anim.currentTime = 750;

CSS Code

@keyframes bounce {
  0%   { transform: translateY(0) scale(1);
         background: #58a6ff; }
  50%  { transform: translateY(-50px) scale(1.2);
         background: #bc8cff; }
  100% { transform: translateY(0) scale(1);
         background: #58a6ff; }
}

.element {
  animation: bounce 1.5s ease-in-out infinite;
}

/* Limited control: */
.paused {
  animation-play-state: paused;
}
/* No reverse, no seek, no speed control,
   no dynamic keyframe modification,
   no finish/cancel events,
   no composite modes */
