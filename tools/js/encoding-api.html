encodeInto() details

const encoder = new TextEncoder(); // encode() - creates new Uint8Array const bytes = encoder.encode('Hello'); // Uint8Array[72, 101, 108, 108, 111] // encodeInto() - writes into existing buffer (efficient) const buf = new Uint8Array(100); const {read, written} = encoder.encodeInto('Hello', buf); // read: 5 chars consumed, written: 5 bytes produced
