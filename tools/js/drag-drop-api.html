API Reference

// Basic drag and drop setup source.addEventListener('dragstart', (e) => { e.dataTransfer.setData('text/plain', 'Hello'); e.dataTransfer.setData('text/html', '<b>Hello</b>'); e.dataTransfer.effectAllowed = 'copyMove'; // Custom drag image const img = new Image(); img.src = 'preview.png'; e.dataTransfer.setDragImage(img, 0, 0); }); target.addEventListener('dragover', (e) => { e.preventDefault(); // Required to allow drop e.dataTransfer.dropEffect = 'copy'; }); target.addEventListener('drop', (e) => { e.preventDefault(); const data = e.dataTransfer.getData('text/plain'); const files = e.dataTransfer.files; // FileList const items = e.dataTransfer.items; // DataTransferItemList const types = e.dataTransfer.types; // string[] }); // Event order: // dragstart -> drag (continuous) -> dragenter -> dragover (continuous) // -> dragleave (if leaving) -> drop -> dragend
