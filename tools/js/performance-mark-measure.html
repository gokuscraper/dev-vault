Code Reference

// Create marks with optional detail (User Timing Level 3) const mark1 = performance.mark('start', { detail: { step: 'initialization', component: 'app' } }); // Create another mark later performance.mark('end', { detail: { step: 'complete' } }); // Measure between marks with detail const measure = performance.measure('load-time', { start: 'start', end: 'end', detail: { category: 'performance' } }); // PerformanceObserver (recommended approach) const observer = new PerformanceObserver((list) => { for (const entry of list.getEntries()) { console.log(entry.name, entry.duration, entry.detail); } }); observer.observe({ type: 'measure', buffered: true });
