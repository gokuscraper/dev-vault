⚙️ Service Worker Integration

// In your service worker (sw.js) // Listen for content deletion by the browser self.addEventListener('contentdelete', (event) => { const id = event.id; event.waitUntil( caches.open('content-v1').then(cache => { // Remove associated cached resources return cache.delete(`/articles/${id}`); }) ); }); // Add content index entry after caching async function cacheAndIndex(content) { const cache = await caches.open('content-v1'); await cache.put(content.url, new Response(content.body)); await self.registration.index.add({ id: content.id, title: content.title, description: content.description, category: 'article', url: content.url, icons: [{ src: '/icons/article-192.png', sizes: '192x192', type: 'image/png' }] }); }

📖 API Quick Reference

// Get the Content Index from a SW registration const reg = await navigator.serviceWorker.ready; const index = reg.index; // Add content await index.add({ id: 'article-1', title: 'My Article', description: 'A great article about PWAs', category: 'article', // 'article' | 'homepage' | 'video' | 'audio' url: '/articles/my-article', icons: [{ src: '/icons/article.png', sizes: '192x192', type: 'image/png' }] }); // List all indexed content const entries = await index.getAll(); entries.forEach(e => console.log(e.id, e.title)); // Delete specific content await index.delete('article-1'); // Browser support check if ('index' in reg) { // Content Index API is supported }
