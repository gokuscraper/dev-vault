👀 Visibility Change Handling

When a tab is hidden, the browser automatically releases the wake lock. You must reacquire it when the tab becomes visible again.
Current visibility: visible
Document hidden: false
document.addEventListener('visibilitychange', async () => {
  if (wakeLock !== null && document.visibilityState === 'visible') {
    wakeLock = await navigator.wakeLock.request('screen');
  }
});
Try switching to another tab and back. Watch the event log for visibility changes and auto-reacquire.

🔋 Battery-Aware Patterns

Consider battery level when deciding whether to acquire a wake lock.
Level: N/A
Charging: N/A
20% 
async function smartWakeLock() {
  if (!('getBattery' in navigator)) {
    return navigator.wakeLock.request('screen');
  }
  const battery = await navigator.getBattery();
  const minLevel = 0.20; // 20%

  if (battery.charging || battery.level > minLevel) {
    return navigator.wakeLock.request('screen');
  }
  console.warn('Battery too low for wake lock');
  return null;
}
