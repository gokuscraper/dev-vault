Worker via Blob URL
// Create a Worker from inline code using a Blob URL
const workerCode = `
  function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
  }

  self.onmessage = (e) => {
    const { algo, n } = e.data;
    const start = performance.now();
    let result;

    if (algo === 'fibonacci') {
      result = fibonacci(n);
    } else if (algo === 'primes') {
      result = sieveOfEratosthenes(n);
    }

    self.postMessage({
      result,
      time: performance.now() - start
    });
  };
`;

const blob = new Blob([workerCode], { type: 'application/javascript' });
const worker = new Worker(URL.createObjectURL(blob));

// Send work to the worker
worker.postMessage({ algo: 'fibonacci', n: 42 });

// Receive results (main thread stays responsive!)
worker.onmessage = (e) => {
  console.log(`Result: ${e.data.result}, Time: ${e.data.time}ms`);
};

// Clean up when done
// worker.terminate();

// Structured Clone (default): data is deep-copied
const buffer = new ArrayBuffer(1024 * 1024 * 10); // 10MB
worker.postMessage({ data: buffer });
console.log(buffer.byteLength); // 10485760 — still accessible

// Transferable: ownership is moved, zero-copy
const buffer2 = new ArrayBuffer(1024 * 1024 * 10);
worker.postMessage({ data: buffer2 }, [buffer2]);
console.log(buffer2.byteLength); // 0 — neutered! Can't use anymore

// Type-safe message protocol
// Main thread:
worker.postMessage({ type: 'COMPUTE', payload: { n: 42 } });
worker.onmessage = (e) => {
  switch (e.data.type) {
    case 'RESULT':
      updateUI(e.data.payload);
      break;
    case 'PROGRESS':
      updateProgressBar(e.data.payload.percent);
      break;
    case 'ERROR':
      showError(e.data.payload.message);
      break;
  }
};

// Worker:
self.onmessage = (e) => {
  const { type, payload } = e.data;
  if (type === 'COMPUTE') {
    for (let i = 0; i < payload.n; i++) {
      // Report progress
      self.postMessage({ type: 'PROGRESS', payload: { percent: i / payload.n } });
    }
    self.postMessage({ type: 'RESULT', payload: { value: 42 } });
  }
};

Worker Pool Pattern

Instead of creating and destroying workers for each task, maintain a pool of reusable workers. Tasks are queued and dispatched to idle workers. This reduces creation overhead and limits resource usage.

Worker Pool Implementation
class WorkerPool {
  constructor(workerScript, size = navigator.hardwareConcurrency || 4) {
    this.size = size;
    this.workers = [];
    this.queue = [];
    this.activeMap = new Map();

    // Create workers
    for (let i = 0; i < size; i++) {
      const worker = new Worker(workerScript);
      worker.id = i;
      this.workers.push({ worker, busy: false });
    }
  }

  // Submit a task and get a Promise back
  exec(data) {
    return new Promise((resolve, reject) => {
      const task = { data, resolve, reject };
      const idle = this.workers.find((w) => !w.busy);

      if (idle) {
        this._dispatch(idle, task);
      } else {
        this.queue.push(task); // Queue if all busy
      }
    });
  }

  _dispatch(slot, task) {
    slot.busy = true;
    this.activeMap.set(slot.worker, task);

    slot.worker.onmessage = (e) => {
      task.resolve(e.data);
      slot.busy = false;
      this.activeMap.delete(slot.worker);

      // Pick next task from queue
      if (this.queue.length > 0) {
        this._dispatch(slot, this.queue.shift());
      }
    };

    slot.worker.onerror = (err) => {
      task.reject(err);
      slot.busy = false;
    };

    slot.worker.postMessage(task.data);
  }

  terminate() {
    this.workers.forEach((w) => w.worker.terminate());
  }
}

// Usage:
const pool = new WorkerPool('compute-worker.js', 4);

const tasks = Array.from({ length: 12 }, (_, i) =>
  pool.exec({ taskId: i, n: 35 + Math.floor(Math.random() * 5) })
);

Promise.all(tasks).then((results) => {
  console.log('All tasks complete', results);
  pool.terminate();
});

SharedWorker & Advanced Concepts

SharedWorkers are shared across all tabs/windows of the same origin. They're useful for shared state, broadcast communication, and resource deduplication. Note: SharedWorker support varies (not available in all browsers).

SharedWorker — Cross-Tab Communication
// shared-worker.js
const ports = new Set();

self.onconnect = (e) => {
  const port = e.ports[0];
  ports.add(port);

  port.onmessage = (event) => {
    const { type, data } = event.data;

    switch (type) {
      case 'BROADCAST':
        // Send to all connected tabs except sender
        ports.forEach((p) => {
          if (p !== port) {
            p.postMessage({ type: 'BROADCAST', data });
          }
        });
        break;

      case 'GET_TAB_COUNT':
        port.postMessage({ type: 'TAB_COUNT', data: ports.size });
        break;
    }
  };

  port.start();

  // Notify all tabs about new connection
  ports.forEach((p) => {
    p.postMessage({ type: 'TAB_COUNT', data: ports.size });
  });
};

// main.js — connecting to SharedWorker
const shared = new SharedWorker('shared-worker.js');
shared.port.start();

shared.port.postMessage({ type: 'BROADCAST', data: 'Hello from tab!' });

shared.port.onmessage = (e) => {
  console.log('Received:', e.data);
};
SharedWorker — Shared WebSocket
// shared-ws-worker.js — one WebSocket shared across all tabs
let ws = null;
const ports = new Set();

function connectWS() {
  ws = new WebSocket('wss://api.example.com/ws');

  ws.onmessage = (event) => {
    // Broadcast to all tabs
    ports.forEach((port) => {
      port.postMessage({ type: 'WS_MESSAGE', data: event.data });
    });
  };

  ws.onclose = () => {
    // Reconnect after 3 seconds
    setTimeout(connectWS, 3000);
  };
}

self.onconnect = (e) => {
  const port = e.ports[0];
  ports.add(port);

  port.onmessage = (event) => {
    if (event.data.type === 'WS_SEND' && ws?.readyState === WebSocket.OPEN) {
      ws.send(event.data.payload);
    }
  };

  port.start();

  // Connect WebSocket on first tab
  if (!ws) connectWS();
};
