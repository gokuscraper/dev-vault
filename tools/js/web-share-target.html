🔄 POST vs GET Method Handling

      
The method field in share_target determines how shared data is sent to your action URL.

    


    

      

        
GET Method

        

          
Data is appended as URL query parameters. Simple but limited — no file sharing support.

          
    
            
  • Data visible in URL
    • 
            
  • URL length limitations
    • 
            
  • No file uploads
    • 
            
  • Easy to implement
    • 
            
  • Works without service worker
    • 
          

        

      

      

        
POST Method

        

          
Data is sent in the request body. Supports files and larger payloads.

          
    
            
  • Data in request body
    • 
            
  • No URL length limits
    • 
            
  • File upload support
    • 
            
  • Requires service worker
    • 
            
  • More secure for sensitive data
    •

📦 Encoding Types

      
When using method: "POST", the enctype field determines how the shared data is encoded in the request body.

    


    

      

        
application/x-www-form-urlencoded

        
Default encoding. Data is sent as key=value pairs, URL-encoded. Cannot handle file uploads.

        
// Request body format:
title=Hello%20World&text=Check%20this&url=https%3A%2F%2F...

// manifest.json
"share_target": {
  "action": "/share",
  "method": "POST",
  "enctype": "application/x-www-form-urlencoded",
  "params": {
    "title": "title",
    "text": "text",
    "url": "url"
  }
}

        
Service Worker Handling

        
self.addEventListener('fetch', (event) => {
  if (event.request.url.includes('/share')) {
    event.respondWith((async () => {
      const formData = await event.request.formData();
      const title = formData.get('title');
      const text = formData.get('text');
      const url = formData.get('url');

      // Process shared data...
      return Response.redirect('/shared?success=true', 303);
    })());
  }
});

      


      

        
multipart/form-data

        
Required for file uploads. Data is sent as multipart boundaries. Use this when accepting shared images, documents, etc.

        
// Request body format:
------WebKitFormBoundary
Content-Disposition: form-data; name="title"

Hello World
------WebKitFormBoundary
Content-Disposition: form-data; name="image";
  filename="photo.jpg"
Content-Type: image/jpeg

[binary data]
------WebKitFormBoundary--

// manifest.json
"share_target": {
  "action": "/share",
  "method": "POST",
  "enctype": "multipart/form-data",
  "params": {
    "title": "title",
    "text": "text",
    "url": "url",
    "files": [{
      "name": "image",
      "accept": ["image/*"]
    }]
  }
}

        
Service Worker Handling

        
self.addEventListener('fetch', (event) => {
  if (event.request.url.includes('/share')) {
    event.respondWith((async () => {
      const formData = await event.request.formData();
      const image = formData.get('image');

      if (image && image.size > 0) {
        // Store in Cache API or IndexedDB
        const cache = await caches.open('shared-files');
        await cache.put('/shared-image',
          new Response(image));
      }

      return Response.redirect('/viewer', 303);
    })());
  }
});

💡 Best Practices

      
Files shared to your PWA are received by the service worker. Store them in Cache API or IndexedDB, then redirect to a page that reads the stored file.

      
Always validate file types and sizes in your service worker. The accept field in the manifest is only a hint to the OS — it does not enforce validation.

      
Use image/* for broad image support, or list specific MIME types like image/png, image/jpeg for strict filtering.

📝 Route Handler Examples

      
Client-side (for GET method)

      
// In your share-handler page's JavaScript:
const params = new URLSearchParams(window.location.search);
const sharedData = {
  title: params.get('title'),
  text: params.get('text'),
  url: params.get('url')
};

// Process the shared data
document.getElementById('shared-title').textContent = sharedData.title;
document.getElementById('shared-text').textContent = sharedData.text;
document.getElementById('shared-url').href = sharedData.url;


      
Service Worker (for POST method)

      
self.addEventListener('fetch', (event) => {
  const url = new URL(event.request.url);

  // Match the share target action URL
  if (url.pathname === '/share-handler'
      && event.request.method === 'POST') {
    event.respondWith(handleShare(event.request));
  }
});

async function handleShare(request) {
  const formData = await request.formData();

  // Store shared data for the client page
  const cache = await caches.open('share-data');
  const sharePayload = JSON.stringify({
    title: formData.get('title'),
    text: formData.get('text'),
    url: formData.get('url'),
    timestamp: Date.now()
  });

  await cache.put('/latest-share',
    new Response(sharePayload, {
      headers: { 'Content-Type': 'application/json' }
    })
  );

  // Redirect to the app page that reads the cache
  return Response.redirect('/?shared=true', 303);
}
