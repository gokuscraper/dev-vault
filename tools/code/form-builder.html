function renderOutput() {
  const area = document.getElementById('outputArea');
  let output = '';
  if (currentTab === 'html') output = genHtml();
  else if (currentTab === 'htmlcss') output = genHtmlCss();
  else if (currentTab === 'react') output = genReact();
  else if (currentTab === 'vue') output = genVue();
  area.textContent = output;
}

function setTab(tab, el) {
  currentTab = tab;
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  renderOutput();
}

function getOutput() {
  if (currentTab === 'html') return genHtml();
  if (currentTab === 'htmlcss') return genHtmlCss();
  if (currentTab === 'react') return genReact();
  if (currentTab === 'vue') return genVue();
  return '';
}

function copyOutput() {
  navigator.clipboard.writeText(getOutput()).then(() => showToast('Copied!'));
}

function downloadOutput() {
  const ext = { html: 'html', htmlcss: 'html', react: 'jsx', vue: 'vue' }[currentTab] || 'txt';
  const blob = new Blob([getOutput()], { type: 'text/plain' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = `form.${ext}`;
  a.click();
  URL.revokeObjectURL(a.href);
}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.display = 'block';
  setTimeout(() => t.style.display = 'none', 2000);
}

function loadPreset(name) {
  fields = [];
  nextId = 1;
  PRESETS[name].forEach(p => {
    const def = JSON.parse(JSON.stringify(FIELD_DEFAULTS[p.type] || {}));
    fields.push({ ...def, ...p, id: nextId++, expanded: false, defaultValue: p.defaultValue || '', pattern: p.pattern || '' });
    if (fields[fields.length-1].type === 'select' || fields[fields.length-1].type === 'radio' || fields[fields.length-1].type === 'checkbox-group') {
      if (!fields[fields.length-1].options) fields[fields.length-1].options = [];
    }
  });
  renderFieldList();
  render();
}
